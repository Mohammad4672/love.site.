<script>

// قلوب نازلة
function قلوب(){
  for(let i=0;i<40;i++){
    let ق=document.createElement("div");
    ق.innerHTML="💖";
    ق.className="قلب";
    ق.style.left=Math.random()*100+"vw";
    ق.style.animationDuration=(2+Math.random()*3)+"s";
    document.body.appendChild(ق);
  }
}

// بداية المفاجأة
function ابدأ(){
  let اسم=document.getElementById("اسم").value;

  if(اسم===""){
    alert("اكتب اسم أول 😏");
    return;
  }

  document.querySelector("h1").innerText="💌 هل تحب/تحبين "+اسم+" ؟";

  document.getElementById("نعم").style.display="inline-block";
  document.getElementById("لا").style.display="inline-block";

  document.body.style.background="linear-gradient(45deg,#ff9a9e,#fad0c4)";

  قلوب();
}

// عند الضغط على نعم
function احب(){
  document.getElementById("رسالة").style.display="block";
  document.body.style.background="linear-gradient(45deg,#ff4d6d,#ffccf9)";
  قلوب();
}

// زر لا يهرب 😂
function اهرب(){
  let زر=document.getElementById("لا");
  زر.style.top=Math.random()*80+"vh";
  زر.style.left=Math.random()*80+"vw";
  زر.innerText="مش لا 😏";
}
